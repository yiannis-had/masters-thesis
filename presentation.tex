\begin{frame}{What is a database?}
  We assume:\pause
  \begin{itemize}
    \item a countably infinite set of values $\mathbb{D}$, called the \emph{domain}.\pause
    \item a totally ordered countably infinite set $\mathcal{U}$ of \emph{attribute names}. We use $U$ to denote a finite subset of $\mathcal{U}$.\pause
  \end{itemize}

  We define:\pause
  \begin{itemize}
    \item A \emph{tuple} as a function $t : U \to \mathbb{D}$.\pause
    \item A \emph{relation} $R$ as a finite set of tuples over $U$.\pause
    \item A \emph{database} $D$ as a finite set of relations.\pause
  \end{itemize}

  \bigskip

  $\{(\textbf{Member\_id}, 4221), (\textbf{First\_name}, \text{Aditya}), (\textbf{Last\_name}, \text{Prakash})\} \in Members$
\end{frame}

\begin{frame}{How do we ask questions?}
  \pause
  \only<2>{\begin{center}\Large\textbf{With queries!}\end{center}}
  \pause
  Loosely, a query takes a database and returns a new relation.

  \pause

  In relational algebra, queries are expressed by applying transformations to relations.\pause

  \bigskip

  The \emph{relational algebra} consists of the following operations:
  \begin{itemize}
    \item Selection: $\sigma_\theta(R)$
    \item Projection: $\pi_L(R)$
    \item Renaming: $\rho_\beta(R)$
    \item Natural join: $R \bowtie S$
    \item Union: $R \cup S$
    \item Difference: $R - S$
  \end{itemize}
\end{frame}

\begin{frame}[t]{When do semirings come in?}
  \pause
  \textit{For all relations $R, S, T$ over attribute set $U$:}\\[0.3cm]
  \begin{center}
    \begin{minipage}{8.2cm}
      \begin{align*}
        R \cup (S \cup T)       & = (R \cup S) \cup T                \\
        R \cup S                & = S \cup R                         \\
        R \cup \mathbf{0}_U     & = R                                \\
        R \bowtie (S \bowtie T) & = (R \bowtie S) \bowtie T          \\
        R \bowtie S             & = S \bowtie R                      \\
        R \bowtie \mathbf{1}    & = R                                \\
        R \bowtie (S \cup T)    & = (R \bowtie S) \cup (R \bowtie T) \\
        R \bowtie \mathbf{0}_U  & = \mathbf{0}_U
      \end{align*}
    \end{minipage}
  \end{center}
\end{frame}

\begin{frame}[t]{What is a (commutative) semiring?}
  \textit{An algebraic structure $(S, +, \cdot, 0, 1)$ such that for all $a,b,c \in S$:}\\[0.3cm]
  \begin{center}
    \begin{minipage}{8.2cm}
      \begin{align*}
        a + (b + c)         & = (a + b) + c               \\
        a + b               & = b + a                     \\
        a + 0               & = a                         \\
        a \cdot (b \cdot c) & = (a \cdot b) \cdot c       \\
        a \cdot b           & = b \cdot a                 \\
        a \cdot 1           & = a                         \\
        a \cdot (b + c)     & = (a \cdot b) + (a \cdot c) \\
        a \cdot 0           & = 0
      \end{align*}
    \end{minipage}
  \end{center}
  \pause
  Examples:
  \begin{itemize}
    \item $\mathbb{N} = (\{0,1,2,\dots\}, +, \times, 0, 1)$
    \item $\mathbb{B} = (\{0, 1\}, \lor, \land, \bot, \top)$
  \end{itemize}
\end{frame}

\begin{frame}{Semimodules to the rescue}
  We need an operation where the values are scaled by the annotations.\pause

  That is exactly what scalar multiplication does: $\ast : S \times M \to M$.\pause

  \bigskip

  Let $S$ be a commutative semiring. An \emph{$S$-semimodule} $(M, +_M, 0_M, \ast)$ is a commutative monoid $(M, +_M, 0_M)$ such that for all $r, s \in S$ and $m, n \in M$:
  \[
    \begin{aligned}
      r \ast (m +_M n)       & = r \ast m +_M r \ast n           \\
      (r +_S s) \ast m       & = r \ast m +_M s \ast m           \\
      (r \cdot_S s) \ast m   & = r \ast (s \ast m)               \\
      1_S \ast m             & = m                               \\
      0_S \ast m = 0_M \quad & \text{and} \quad r \ast 0_M = 0_M
    \end{aligned}
  \]
\end{frame}
